# Create a singly linked list and traverse it
 
class Node:
  def __init__(self, data):
      self.data = data
      self.next = None
 
class LinkedList:
  def __init__(self):
      self.head = None
   def insert_at_end(self, data):
      new_node = Node(data)
      if self.head is None:
          self.head = new_node
      else:
          temp = self.head
          while(temp.next):
              temp = temp.next
          temp.next = new_node
   def traverse(self):
      temp = self.head
      while(temp):
          print(temp.data)
          temp = temp.next

# Create a graph and traverse it
 
from collections import defaultdict
 
graph = defaultdict(list)
graph[1].append(2) # 1 -> 2
graph[2].append(3) # 1 -> 2 -> 3
graph[4].append(1) # 4 -> 1 -> 2 -> 3
 
visited = set()
 
def dfs(node, graph, visited):
  if node not in visited:
      print(node)
      visited.add(node)
      for v in graph[node]:
          dfs(v, graph, visited)
 
dfs(4, graph, visited)
